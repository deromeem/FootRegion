SELECT m.id, m.nom, m.date_heure, m.score_domicile, m.score_invite, m.tournois_id, equipes_domicile_id, equipes_invite_id, m.adr_rue, m.adr_ville, m.adr_cp, m.coord_gps, e.nom AS equipe_invite,ed.nom AS equipe_domicile,einv.email AS entraineur_invite,ein.email AS entraineur_initiateur,t.nom AS tournoi,s.statut AS statut
FROM footregion_footregion_matchs m
LEFT JOIN footregion_footregion_equipes AS e ON e.id=m.equipes_invite_id
LEFT JOIN footregion_footregion_equipes AS ed ON ed.id=m.equipes_domicile_id
LEFT JOIN footregion_footregion_entraineurs AS einv ON einv.id=m.entraineurs_invite_id
LEFT JOIN footregion_footregion_entraineurs AS ein ON ein.id=m.entraineurs_initiateur_id
LEFT JOIN footregion_footregion_tournois AS t ON t.id=m.tournois_id
LEFT JOIN footregion_footregion_statuts AS s ON s.id=m.statuts_id
WHERE (m.published=0 OR m.published=1)
ORDER BY m.date_heure desc;
